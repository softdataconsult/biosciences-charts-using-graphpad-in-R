# THEME
# ============================================================
bw_theme <- theme_classic() +
  theme(
    panel.border    = element_rect(colour = "black", fill = NA, linewidth = 0.8),
    axis.line       = element_blank(),
    axis.text.x     = element_text(colour = "black", size = 9,
                                   angle = 45, hjust = 1),   # <-- slanted labels
    axis.text.y     = element_text(colour = "black", size = 9),
    axis.title      = element_text(colour = "black", size = 10),
    plot.title      = element_text(size = 8, hjust = 0.5, face = "italic"),
    legend.position = "none",
    axis.ticks      = element_line(colour = "black")
  )

# ============================================================
# PLOT FUNCTION
# ============================================================
make_plot <- function(assay_name) {
  df        <- data_list[[assay_name]]
  df$group  <- factor(df$group, levels = groups)
  df$label_y <- df$mean + df$sd + max(df$mean) * 0.06

  ggplot(df, aes(x = group, y = mean)) +
    geom_bar_pattern(
      aes(pattern_angle   = group,
          pattern_spacing = group),
      stat            = "identity",
      pattern         = "stripe",
      fill            = "white",
      colour          = "black",
      linewidth       = 0.5,
      pattern_colour  = "black",
      pattern_fill    = "black",
      pattern_density = 0.4,
      width           = 0.65
    ) +
    scale_pattern_angle_manual(
      values = c("Group 1" = 45,  "Group 2" = 0,
                 "Group 3" = 90,  "Group 4" = 135, "Group 5" = 45)
    ) +
    scale_pattern_spacing_manual(
      values = c("Group 1" = 0.07, "Group 2" = 0.05,
                 "Group 3" = 0.05, "Group 4" = 0.07, "Group 5" = 0.03)
    ) +
    geom_errorbar(
      aes(ymin = mean - sd, ymax = mean + sd),
      width = 0.2, linewidth = 0.6, colour = "black"
    ) +
    geom_text(
      aes(y = label_y, label = letter),
      size = 3.5, fontface = "italic", vjust = 0
    ) +
    scale_y_continuous(expand = expansion(mult = c(0, 0.15))) +
    labs(
      x     = "Treatment",
      y     = ylabels[[assay_name]],
      title = titles[[assay_name]]
    ) +
    bw_theme
}

# SAVE INDIVIDUAL PLOTS
# ============================================================
assay_names <- names(data_list)

for (a in assay_names) {
  p <- make_plot(a)
  ggsave(

    plot     = p,
    width    = 5, height = 4.2, dpi = 300, bg = "white"
  )
  cat("Saved:", a, "\n")
}
